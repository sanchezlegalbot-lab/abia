          {area.items.map((item, i) => (
            <button key={item.id} onClick={() => setSelected(item)}
              onMouseEnter={() => setHov(item.id)} onMouseLeave={() => setHov(null)}
              style={{ width: "100%", background: hov === item.id ? area.color + "0f" : "transparent", border: "none", borderBottom: i < area.items.length - 1 ? "1px solid #f0f0ee" : "none", padding: "13px 26px", display: "flex", alignItems: "center", justifyContent: "space-between", cursor: "pointer", textAlign: "left" }}>
              <div style={{ display: "flex", alignItems: "center", gap: 11 }}>
                <div style={{ width: 6, height: 6, borderRadius: "50%", background: hov === item.id ? area.color : "#ddd", flexShrink: 0 }} />
                <span style={{ fontSize: 13.5, fontWeight: hov === item.id ? 600 : 400, color: hov === item.id ? area.color : "#333" }}>{item.name}</span>
              </div>
              <span style={{ color: hov === item.id ? area.color : "#ddd", fontSize: 16 }}>›</span>
            </button>
          ))}
        </div>
      </div>
      {selected && <ItemModal item={selected} color={area.color} onClose={() => setSelected(null)} />}
    </>
  );
}

function Login({ onLogin }) {
  const [usuario, setUsuario] = useState("");
  const [clave, setClave] = useState("");
  const [error, setError] = useState("");
  const [verClave, setVerClave] = useState(false);
  const [loading, setLoading] = useState(false);

  const handleLogin = () => {
    setError("");
    setLoading(true);
    setTimeout(() => {
      const ok = USUARIOS.find(u => u.usuario === usuario.trim() && u.clave === clave);
      if (ok) { onLogin(ok); }
      else { setError("Usuario o contraseña incorrectos."); setLoading(false); }
    }, 600);
  };

  return (
    <div style={{ minHeight: "100vh", background: "#f4f3f0", display: "flex", alignItems: "center", justifyContent: "center", fontFamily: "DM Sans, sans-serif" }}>
      <div style={{ width: "100%", maxWidth: 420, padding: "0 16px" }}>
        <div style={{ background: "#fff", borderRadius: 4, overflow: "hidden", boxShadow: "0 8px 40px rgba(0,0,0,0.10)" }}>
          <div style={{ background: BLUE, padding: "36px 36px 28px", position: "relative", overflow: "hidden" }}>
            <Curvas opacity={0.28} />
            <div style={{ position: "relative", zIndex: 1 }}>
              <LogoABI />
              <div style={{ marginTop: 18, fontSize: 13, color: "rgba(255,255,255,0.6)" }}>Tu asistente legal inteligente</div>
            </div>
            <div style={{ position: "absolute", bottom: 0, left: 0, right: 0, height: 3, background: `linear-gradient(90deg, transparent, ${ORANGE})` }} />
          </div>
          <div style={{ padding: "32px 36px 36px" }}>
            <div style={{ fontSize: 17, fontWeight: 600, color: "#111", marginBottom: 6 }}>Iniciar sesión</div>
            <div style={{ fontSize: 13, color: "#999", marginBottom: 26 }}>Ingresa tus credenciales para continuar</div>
            <div style={{ marginBottom: 14 }}>
              <label style={{ fontSize: 11, fontWeight: 600, color: "#555", letterSpacing: 0.5, textTransform: "uppercase", display: "block", marginBottom: 6 }}>Usuario</label>
              <input value={usuario} onChange={e => setUsuario(e.target.value)} onKeyDown={e => e.key === "Enter" && handleLogin()} placeholder="tu.usuario"
                style={{ width: "100%", padding: "11px 14px", borderRadius: 2, border: `1.5px solid ${error ? "#ef4444" : "#e5e5e3"}`, fontSize: 14, outline: "none", boxSizing: "border-box", background: "#fafaf9" }}
                onFocus={e => e.target.style.borderColor = BLUE} onBlur={e => e.target.style.borderColor = error ? "#ef4444" : "#e5e5e3"} />
            </div>
            <div style={{ marginBottom: 22 }}>
              <label style={{ fontSize: 11, fontWeight: 600, color: "#555", letterSpacing: 0.5, textTransform: "uppercase", display: "block", marginBottom: 6 }}>Contraseña</label>
              <div style={{ position: "relative" }}>
                <input type={verClave ? "text" : "password"} value={clave} onChange={e => setClave(e.target.value)} onKeyDown={e => e.key === "Enter" && handleLogin()} placeholder="••••••••"
                  style={{ width: "100%", padding: "11px 40px 11px 14px", borderRadius: 2, border: `1.5px solid ${error ? "#ef4444" : "#e5e5e3"}`, fontSize: 14, outline: "none", boxSizing: "border-box", background: "#fafaf9" }}
                  onFocus={e => e.target.style.borderColor = BLUE} onBlur={e => e.target.style.borderColor = error ? "#ef4444" : "#e5e5e3"} />
                <button onClick={() => setVerClave(!verClave)} style={{ position: "absolute", right: 11, top: "50%", transform: "translateY(-50%)", background: "none", border: "none", cursor: "pointer", color: "#aaa", fontSize: 14 }}>
                  {verClave ? "🙈" : "👁"}
                </button>
              </div>
              {error && <div style={{ fontSize: 12, color: "#ef4444", marginTop: 5 }}>{error}</div>}
            </div>
            <button onClick={handleLogin} disabled={loading}
              style={{ width: "100%", padding: "13px", background: loading ? "#8fa3f0" : BLUE, border: "none", borderRadius: 2, color: "#fff", fontSize: 14, fontWeight: 600, cursor: loading ? "default" : "pointer", display: "flex", alignItems: "center", justifyContent: "center", gap: 8 }}
              onMouseEnter={e => { if (!loading) e.currentTarget.style.background = "#1728b8"; }}
              onMouseLeave={e => { if (!loading) e.currentTarget.style.background = BLUE; }}>
              {loading ? "Verificando..." : "Ingresar →"}
            </button>
          </div>
        </div>
        <div style={{ textAlign: "center", marginTop: 16, fontSize: 11, color: "#bbb" }}>
          Desarrollado por <strong style={{ color: "#aaa" }}>Tomás Sánchez E.</strong> · Claude by Anthropic
        </div>
      </div>
    </div>
  );
}
